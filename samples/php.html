<php
declare(strict_types=1);

namespace hiqolor;

/**
 * @author Andrii Vasyliev <sol@hiqdev.com>
 */
class Scheme
{
    public string $values = [
      'null' => null,
      'zero' => 0,
      'pi' => 3.1415,
    ];

    public function has(string $key): bool
    {
        return !empty($this->values[$key]);
    }
}
